import Mathlib
open scoped BigOperators
namespace HeadComplexity
open MvPolynomial
variable {n : ℕ}

/-- Map a Boolean assignment to a real point of the cube `{0,1}ⁿ ⊆ ℝⁿ`. -/
def boolToReal (x : Fin n → Bool) : Fin n → ℝ := fun i => if x i then 1 else 0

/-- A real polynomial `P` *sign-represents* `f` on the Boolean cube if the sign of
its evaluation at each cube point matches the truth value of `f`. Encoded as a
strict positivity ⇔ `f = true`. -/
def SignRepresents (P : MvPolynomial (Fin n) ℝ) (f : (Fin n → Bool) → Bool) : Prop :=
  ∀ x : Fin n → Bool, (0 < MvPolynomial.eval (boolToReal x) P ↔ f x = true)

/-- `f` admits a sign representation of total degree `≤ d`. -/
def SignRepDegLE (f : (Fin n → Bool) → Bool) (d : ℕ) : Prop :=
  ∃ P : MvPolynomial (Fin n) ℝ, P.totalDegree ≤ d ∧ SignRepresents P f

/-- Indicator polynomial of a cube point `a`: `∏ i, (Xᵢ if aᵢ else 1-Xᵢ)`. -/
noncomputable def indicator (a : Fin n → Bool) : MvPolynomial (Fin n) ℝ :=
  ∏ i : Fin n, (if a i then X i else 1 - X i)

lemma totalDegree_factor_le (a : Fin n → Bool) (i : Fin n) :
    ((if a i then (X i : MvPolynomial (Fin n) ℝ) else 1 - X i)).totalDegree ≤ 1

lemma totalDegree_indicator_le (a : Fin n → Bool) : (indicator a).totalDegree ≤ n

lemma eval_factor (a y : Fin n → Bool) (i : Fin n) :
    MvPolynomial.eval (boolToReal y)
        (if a i then (X i : MvPolynomial (Fin n) ℝ) else 1 - X i)
      = (if a i = y i then 1 else 0)

lemma eval_indicator (a y : Fin n → Bool) :
    MvPolynomial.eval (boolToReal y) (indicator a) = (if a = y then 1 else 0)

/-- Sign-representing multilinear interpolation of `f`. -/
noncomputable def interp (f : (Fin n → Bool) → Bool) : MvPolynomial (Fin n) ℝ :=
  ∑ a : Fin n → Bool, (if f a then (1 : ℝ) else -1) • indicator a

lemma totalDegree_interp_le (f : (Fin n → Bool) → Bool) : (interp f).totalDegree ≤ n

lemma eval_interp (f : (Fin n → Bool) → Bool) (y : Fin n → Bool) :
    MvPolynomial.eval (boolToReal y) (interp f) = (if f y then (1 : ℝ) else -1)

theorem SignRepresents_interp (f : (Fin n → Bool) → Bool) : SignRepresents (interp f) f

/-- Trivial existence: every Boolean `f` is sign-representable in degree `≤ n`. -/
theorem signRep_exists (f : (Fin n → Bool) → Bool) : SignRepDegLE f n

end HeadComplexity

import Mathlib

open scoped BigOperators

namespace HeadComplexity

open MvPolynomial

variable {n : ℕ}

/-- Map a Boolean assignment to a real point of the cube `{0,1}ⁿ`. -/
def boolToReal (x : Fin n → Bool) : Fin n → ℝ := fun i => if x i then 1 else 0

/-- A real polynomial `P` *sign-represents* `f` on the Boolean cube if its sign at
each cube point matches the truth value of `f`. -/
def SignRepresents (P : MvPolynomial (Fin n) ℝ) (f : (Fin n → Bool) → Bool) : Prop :=
  ∀ x : Fin n → Bool, (0 < MvPolynomial.eval (boolToReal x) P ↔ f x = true)

/-- `f` has a degree-`≤ d` sign representation. -/
def SignRepDegLE (f : (Fin n → Bool) → Bool) (d : ℕ) : Prop :=
  ∃ P : MvPolynomial (Fin n) ℝ, P.totalDegree ≤ d ∧ SignRepresents P f

/-- The indicator polynomial of a cube point `a`. -/
noncomputable def indicator (a : Fin n → Bool) : MvPolynomial (Fin n) ℝ :=
  ∏ i : Fin n, (if a i then X i else 1 - X i)

lemma totalDegree_factor_le (a : Fin n → Bool) (i : Fin n) :
    ((if a i then (X i : MvPolynomial (Fin n) ℝ) else 1 - X i)).totalDegree ≤ 1 := by
  by_cases h : a i
  · simp [h, totalDegree_X]
  · simp only [h]
    calc (1 - X i : MvPolynomial (Fin n) ℝ).totalDegree
        ≤ max (1 : MvPolynomial (Fin n) ℝ).totalDegree (X i).totalDegree := totalDegree_sub _ _
      _ ≤ 1 := by rw [totalDegree_one, totalDegree_X]; simp

lemma totalDegree_indicator_le (a : Fin n → Bool) :
    (indicator a).totalDegree ≤ n := by
  unfold indicator
  calc (∏ i : Fin n, (if a i then (X i : MvPolynomial (Fin n) ℝ) else 1 - X i)).totalDegree
      ≤ ∑ i : Fin n, ((if a i then (X i : MvPolynomial (Fin n) ℝ) else 1 - X i)).totalDegree :=
        totalDegree_finset_prod _ _
    _ ≤ ∑ _i : Fin n, 1 := Finset.sum_le_sum (fun i _ => totalDegree_factor_le a i)
    _ = n := by simp

lemma eval_factor (a y : Fin n → Bool) (i : Fin n) :
    MvPolynomial.eval (boolToReal y) (if a i then (X i : MvPolynomial (Fin n) ℝ) else 1 - X i)
      = (if a i = y i then 1 else 0) := by
  by_cases ha : a i <;> by_cases hy : y i <;>
    simp [ha, hy, boolToReal, eval_sub]

lemma eval_indicator (a y : Fin n → Bool) :
    MvPolynomial.eval (boolToReal y) (indicator a) = (if a = y then 1 else 0) := by
  unfold indicator
  rw [eval_prod]
  have : (∏ i : Fin n, MvPolynomial.eval (boolToReal y)
      (if a i then (X i : MvPolynomial (Fin n) ℝ) else 1 - X i))
      = ∏ i : Fin n, (if a i = y i then (1 : ℝ) else 0) := by
    apply Finset.prod_congr rfl
    intro i _; exact eval_factor a y i
  rw [this]
  by_cases h : a = y
  · subst h; simp
  · rw [if_neg h]
    rw [Finset.prod_eq_zero_iff]
    obtain ⟨i, hi⟩ := Function.ne_iff.mp h
    exact ⟨i, Finset.mem_univ i, by simp [hi]⟩

/-- The sign-representing interpolation polynomial of `f`. -/
noncomputable def interp (f : (Fin n → Bool) → Bool) : MvPolynomial (Fin n) ℝ :=
  ∑ a : Fin n → Bool, (if f a then (1 : ℝ) else -1) • indicator a

lemma totalDegree_interp_le (f : (Fin n → Bool) → Bool) :
    (interp f).totalDegree ≤ n := by
  unfold interp
  refine totalDegree_finsetSum_le (fun a _ => ?_)
  exact (totalDegree_smul_le _ _).trans (totalDegree_indicator_le a)

lemma eval_interp (f : (Fin n → Bool) → Bool) (y : Fin n → Bool) :
    MvPolynomial.eval (boolToReal y) (interp f) = (if f y then (1 : ℝ) else -1) := by
  unfold interp
  rw [eval_sum]
  have : (∑ a : Fin n → Bool, MvPolynomial.eval (boolToReal y)
      ((if f a then (1 : ℝ) else -1) • indicator a))
      = ∑ a : Fin n → Bool, (if a = y then (if f a then (1:ℝ) else -1) else 0) := by
    apply Finset.sum_congr rfl
    intro a _
    rw [smul_eval, eval_indicator]
    by_cases h : a = y <;> simp [h]
  rw [this, Finset.sum_ite_eq' Finset.univ y (fun a => if f a then (1:ℝ) else -1)]
  simp

theorem SignRepresents_interp (f : (Fin n → Bool) → Bool) :
    SignRepresents (interp f) f := by
  intro x
  rw [eval_interp]
  by_cases h : f x = true
  · simp [h]
  · simp only [Bool.not_eq_true] at h
    simp [h]

theorem signRep_exists (f : (Fin n → Bool) → Bool) : SignRepDegLE f n :=
  ⟨interp f, totalDegree_interp_le f, SignRepresents_interp f⟩

end HeadComplexity
